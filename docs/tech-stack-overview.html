Quick Start Commands

Start Everything

# Start all services
./start_all.sh

# Or separately:
./start_backend.sh   # Backend on http://localhost:8000
./start_frontend.sh  # Frontend on http://localhost:5173

Development

# Backend with hot reload
cd apps/api && uvicorn main:app --reload --port 8000

# Frontend dev server
cd apps/web && npm run dev

Database

# Seed database with sample data
python tools/setup/seed_database.py

# Run migrations
python tools/migrations/run_migrations.py

# Setup RAG vector store
python tools/setup/setup_rag.py

Testing

# Run all tests
cd tools/testing && python run_all_tests.py

# Run specific test suites
python -m pytest tools/testing/test_all_agents.py -v
python -m pytest tools/testing/test_api_endpoints.py -v
python -m pytest tools/testing/test_integration.py -v

Lint & Format

# Frontend linting
cd apps/web && npm run lint

# Backend formatting
cd apps/api && black . && isort .

Celery Worker Optional - Requires Redis

# Requires Redis running first: redis-server
# Start Celery worker (optional - app works without it)
cd apps/api && celery -A celery_app worker --loglevel=info

# Flower monitoring (not yet installed - planned)
# pip install flower && celery -A celery_app flower --port=5555
